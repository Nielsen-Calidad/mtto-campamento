Procedimiento de referencia · PO-MTTO-005

Las OS se generan mediante solicitud del área correspondiente. El responsable de logística registra y asigna número de OS. La codificación es OS-XX-YYYY donde XX indica el área destino (MT: Taller Metalúrgico · CA: Campamento) e YYYY es la numeración correlativa.

CONFIGURACIÓN

Integración con Google Sheets · sincronización en la nube

📋 Instrucciones — Google Apps Script

1. En tu Google Drive, creá una nueva Hoja de Cálculo (Google Sheets) con el nombre que quieras.
2. En el menú Extensiones → Apps Script, pegá el siguiente código y hacé clic en Implementar → Nueva implementación → Aplicación web.
3. En "Quién tiene acceso" seleccioná Cualquier persona y copiá la URL generada.
4. Pegá esa URL en el campo de arriba.

const SHEET_NAME = "OS"; const USERS_SHEET = "Usuarios"; // ── Utilidad: hoja a array de objetos ── function sheetToObjects(sheet) { const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); return obj; }); } // ── Hoja Usuarios: inicializar si no existe ── function getOrCreateUsersSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(USERS_SHEET); if (!sheet) { sheet = ss.insertSheet(USERS_SHEET); const cols = ["id","username","password","name","role","area"]; sheet.getRange(1,1,1,cols.length).setValues([cols]); const defaults = [ ["u1","admin", "Nielsen2025!","Administrador","admin", "Administracion"], ["u2","responsable", "Mtto2025!", "Resp. MTTO", "responsable", "Mantenimiento"], ["u3","tecnico", "Tec2025!", "Tecnico", "tecnico", "Taller Metalurgico"], ["u4","solicitante", "Sol2025!", "Solicitante", "solicitante", "Campamento"] ]; sheet.getRange(2,1,defaults.length,cols.length).setValues(defaults); } return sheet; } function doGet(e) { const action = e && e.parameter && e.parameter.action; // ── GET usuarios ── if (action === "getUsers") { const sheet = getOrCreateUsersSheet(); return ContentService .createTextOutput(JSON.stringify(sheetToObjects(sheet))) .setMimeType(ContentService.MimeType.JSON); } // ── GET órdenes (default) ── const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); return ContentService .createTextOutput(JSON.stringify(sheetToObjects(sheet))) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const body = JSON.parse(e.postData.contents); // ── Guardar usuarios ── if (body.action === "saveUsers") { const sheet = getOrCreateUsersSheet(); const cols = ["id","username","password","name","role","area"]; sheet.clearContents(); sheet.getRange(1,1,1,cols.length).setValues([cols]); if (body.users && body.users.length) { const rows = body.users.map(u => cols.map(c => u[c] || "")); sheet.getRange(2,1,rows.length,cols.length).setValues(rows); } return ContentService.createTextOutput("ok"); } // ── OS: eliminar ── const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); if (body.action === "delete") { const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == body.id) { sheet.deleteRow(i+1); break; } } return ContentService.createTextOutput("ok"); } // ── OS: crear / actualizar ── let headers = sheet.getLastColumn() > 0 ? sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0] : []; if (!headers[0]) { const cols = ["id","numero","fecha_pedido","tipo","solicitante","area_solicitante", "responsable","area_destino","plazo","fecha_inicio","fecha_fin","duracion_hs", "desc_trabajo","trabajo_realizado","observaciones","materiales","mano_obra","estado","obs_final"]; sheet.getRange(1,1,1,cols.length).setValues([cols]); headers = cols; } const data = sheet.getDataRange().getValues(); const row = headers.map(h => body[h] !== undefined ? body[h] : ""); for (let i = 1; i < data.length; i++) { if (data[i][0] == body.id) { sheet.getRange(i+1,1,1,row.length).setValues([row]); return ContentService.createTextOutput("updated"); } } sheet.appendRow(row); return ContentService.createTextOutput("created"); }

Flujo de aprobación — PO-MTTO-005 sección 6.7 y 6.8

Una vez finalizado el trabajo, el Responsable de MTTO aprueba y firma la OS. Luego el ejecutante firma y finalmente el solicitante da conformidad del servicio. Solo entonces la OS se considera cerrada (estado: Aprobada).
